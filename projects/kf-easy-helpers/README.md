
## Intalling
Run `npm i kf-easy-helpers --save`  to install in your project.


import { KfEasyHelpersModule } from 'kf-easy-helpers'

@NgModule({
    declarations: [
        AppComponent,
    ],
    imports: [
        KfEasyHelpersModule 
    ],
    providers: [
    ],
    bootstrap: [AppComponent]
})
export class AppModule { }
